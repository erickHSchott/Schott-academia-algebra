    <script>
        const questions = [
            { q: "¿Cómo se escribe 'un número aumentado en 5'?", a: ["x - 5", "x + 5", "5x", "x / 5"], c: 1, r: "Aumentar significa sumar (+)." },
            { q: "Si 'x' es tu edad, ¿cómo escribirías la edad que tendrás en 3 años?", a: ["x + 3", "3x", "x - 3", "3 - x"], c: 0, r: "El futuro siempre es sumando años." },
            { q: "Traduce: 'El triple de un número'", a: ["x + 3", "x³", "3x", "3 - x"], c: 2, r: "Triple es multiplicar por 3, y en álgebra se ponen juntos." }
        ];

        let current = 0;
        let score = 0;

        function showView(viewId) {
            document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
            document.getElementById(viewId).classList.add('active');
        }

        function startQuiz() {
            current = 0;
            score = 0;
            showView('view-quiz');
            loadQuestion();
        }

        function loadQuestion() {
            const q = questions[current];
            document.getElementById('question').innerText = q.q;
            const optDiv = document.getElementById('options');
            optDiv.innerHTML = '';
            
            q.a.forEach((opt, i) => {
                const b = document.createElement('div');
                b.className = 'option';
                b.innerText = opt;
                b.onclick = () => check(i, b);
                optDiv.appendChild(b);
            });
            document.getElementById('next-btn').style.display = 'none';
            document.getElementById('progress-bar').style.width = (current / questions.length * 100) + "%";
        }

        function check(idx, el) {
            const q = questions[current];
            if(idx === q.c) {
                score++;
                el.classList.add('selected-correct');
                confetti({ particleCount: 40, spread: 60 });
            } else {
                el.classList.add('selected-wrong');
                document.querySelectorAll('.option')[q.c].classList.add('selected-correct');
            }
            document.querySelectorAll('.option').forEach(o => o.style.pointerEvents = 'none');
            document.getElementById('next-btn').style.display = 'block';
        }

        function nextQuestion() {
            current++;
            if(current < questions.length) loadQuestion();
            else showFinalResults();
        }

        function showFinalResults() {
            showView('view-results');
            document.getElementById('score-display').innerText = score + "/" + questions.length;
            document.getElementById('result-title').innerText = score > 1 ? "¡Excelente Trabajo! 🏆" : "¡Sigue practicando! 💪";
        }
    </script>
